73char calcChecksum(ifstream* inputFile, bool timeDependent) {
74 if (!inputFile->is_open() || !inputFile->good()) {
75 return 1; // Ошибка открытия файла
76 }
77
78 char ch;
79 unsigned long long checksum = 0;
80 unsigned long long sdvig = 0;
81 unsigned long long index = 0; // Индекс для определения позиции символа
82
83 while (inputFile->get(ch)) {
84 // Определяем сдвиг в зависимости от символа
85 if (isalpha(ch)) { // Латинские буквы
86 sdvig = (ch >= L'a' && ch <= L'z') ? 3 : 3; // Латинские буквы - сдвиг 3
87 } else if ((ch >= 0xC0 && ch <= 0xFF) || (ch >= 0xE0 && ch <= 0xEF)) { // Кириллица в UTF-8
88 sdvig = 5; // Кириллица - сдвиг 5
89 } else if (isdigit(ch)) { // Цифры
90 sdvig = 2; // Цифры - сдвиг 2
91 } else {
92 sdvig = 0; // Прочие символы
93 }
94
95 // Применяем сдвиг и обновляем контрольную сумму, учитывая индекс символа
96 checksum += (ch + sdvig) * (index + 1); // Увеличиваем значение в зависимости от позиции
97 index++; // Увеличиваем индекс
98 }
99
100 if (timeDependent) {
101 // Добавляем текущее время к контрольной сумме
102 checksum += get_minute() % 2; // Пример
103 }
104
105 inputFile->clear();
106 inputFile->seekg(0, ios_base::beg);
107
108 return (char)checksum;
109}
